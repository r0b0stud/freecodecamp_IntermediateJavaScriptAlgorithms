function addTogether() {
  var n1=arguments[0];
  var n2=arguments[1];
  if(arguments.length==2){
    if(
      typeof arguments[0]==='number'
      && typeof arguments[1]==='number'
      ){
        return(arguments[0]+arguments[1]);
        }
    else{
      return undefined;
    }
  }
  else{
    if(typeof arguments[0] !== 'number'){
      return undefined
    }
    else{
      return function(){
        if(typeof arguments[0] !== 'number'){
          return undefined
        }
        else{
          return arguments[0]+n1;
        }
      }
    }
  }
}
